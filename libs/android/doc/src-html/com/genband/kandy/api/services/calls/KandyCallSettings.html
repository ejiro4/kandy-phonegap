001package com.genband.kandy.api.services.calls;
002
003import com.genband.kandy.api.services.common.KandyCameraInfo;
004import com.genband.kandy.api.utils.KandyIllegalArgumentException;
005import com.genband.kandy.core.common.KandyStrings;
006import com.genband.kandy.providers.route.services.RouteRegistrationCoreService;
007import com.genband.mobile.api.utilities.Configuration;
008
009public class KandyCallSettings {
010
011        public KandyCameraInfo getCameraMode()
012        {
013                KandyCameraInfo cameraInfo = null;
014                Configuration config = RouteRegistrationCoreService.getInstance().getSettings().getSpidrConfiguration();
015                if (config != null)
016                {
017                        cameraInfo = KandyCameraInfo.fromInt(config.getDefaultCameraMode());
018                }
019                return cameraInfo == null ? getDefaultCameraMode() : cameraInfo;
020        }
021
022        public void setCameraMode(KandyCameraInfo newMode) throws KandyIllegalArgumentException
023        {
024                if (newMode == null)
025                {
026                        throw new KandyIllegalArgumentException(KandyStrings.ERROR_MSG_INVALID_ARGUMENT);
027                }
028                
029                Configuration config = RouteRegistrationCoreService.getInstance().getSettings().getSpidrConfiguration();
030                if (config != null)
031                {
032                        config.setDefaultCameraMode(newMode.toValue());
033                }
034        }
035
036        private KandyCameraInfo getDefaultCameraMode()
037        {
038                return KandyCameraInfo.FACING_FRONT;
039        }
040}
