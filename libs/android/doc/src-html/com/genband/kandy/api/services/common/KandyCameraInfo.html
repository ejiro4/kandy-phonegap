001package com.genband.kandy.api.services.common;
002
003import android.hardware.Camera.CameraInfo;
004
005public enum KandyCameraInfo
006{
007        UNKNOWN(-1),
008        /**
009         * The facing of the camera is opposite to that of the screen.
010         * 
011         * @see {@link CameraInfo#CAMERA_FACING_BACK}
012         */
013        FACING_BACK(0),
014
015        /**
016         * The facing of the camera is the same as that of the screen.
017         * 
018         * @see {@link CameraInfo#CAMERA_FACING_FRONT}
019         */
020        FACING_FRONT(1);
021
022        private int mValue = -1;
023
024        KandyCameraInfo(int value)
025        {
026                this.mValue = value;
027        }
028
029        public int toValue()
030        {
031                return this.mValue;
032        }
033        
034        public static KandyCameraInfo fromInt(int value)
035        {
036                for (KandyCameraInfo info : KandyCameraInfo.values())
037                {
038                        if (info.toValue() == value)
039                        {
040                                return info;
041                        }
042                }
043                
044                return null;
045        }
046}
