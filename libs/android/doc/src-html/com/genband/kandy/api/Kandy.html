022/**
023 * Use this class for all the following functionality:
024 * <ul>
025 * <li>{@link IKandyProvisioning}
026 * <li>{@link IKandyAccess}
027 * <li>{@link IKandyServices}
028 * <ul>
029 * <li>{@link IKandyCallService}
030 * <li>{@link IKandyChatService}
031 * <li>{@link IKandyAccess}
032 * <li>{@link IKandyPresence}
033 * <li>{@link IKandyLocationService}
034 * <li>{@link IKandyPushService}
035 * </ul>
036 * 
037 */
038public class Kandy
039{
040        private static final String TAG = Kandy.class.getSimpleName();
041
042        private static Kandy sInstance = null;
043
044        private static Context sContext;
045
046        private Kandy()
047        {
048                KandyLog.d(TAG, "initialize " + TAG);
049        }
050
051        public static synchronized Kandy getInstance()
052        {
053                if (sInstance == null)
054                {
055                        sInstance = new Kandy();
056                }
057                return sInstance;
058        }
059
060        /**
061         * Initialize the Kandy SDK with your Api key & secret.
062         * <p>
063         * <b>Note:</b> If you need to create your keys go to <a
064         * href="http://www.kandy.io/">Kandy</a>
065         * 
066         * @param context The appliction Contexts
067         * @param apiKey Your api key genereted from <a href="http://www.kandy.io/my-kandy">My Kandy</a> 
068         * @param apiSecret Your api secret genereted from <a href="http://www.kandy.io/my-kandy">My Kandy</a>
069         */
070        public static void initialize(Context context, String apiKey, String apiSecret)
071        {
072                Kandy.sContext = context;
073
074                // log SDK version
075                KandyLog.d(TAG, "Kandy: " + "####################################");
076                KandyLog.d(TAG, "Kandy: " + "#### SDK version: " + KandyConfig.KANDY_SDK_VERSOIN_NAME + "("
077                                + KandyConfig.KANDY_SDK_VERSOIN_CODE + ") ####");
078                KandyLog.d(TAG, "Kandy: " + "#### SDK environment: " + KandyConfig.BUILD_ENVIRONMENT.toString() + " ##");
079                KandyLog.d(TAG, "Kandy: " + "####################################");
080
081                KandyLog.d(TAG, "initialize: ApiKey: " + apiKey + " ApiSecret: " + apiSecret);
082
083                // KandyPreferencesManager.getInstance().putString(KandyKeys.KEY_API_KEY,
084                // apiKey);
085                // KandyPreferencesManager.getInstance().putString(KandyKeys.KEY_API_SECRET,
086                // apiSecret);
087                KandyDomain kandyDomain = new KandyDomain();
088                kandyDomain.setApiKey(apiKey);
089                kandyDomain.setApiSecret(apiSecret);
090                KandyCore.getInstance().getSession().setKandyDomain(kandyDomain);
091
092                init();
093        }
094
095        private static void init()
096        {
097                // init the main Event Handler
098                IKandyEventHandler handler = KandyCore.getInstance().getEventsHandler();
099                // set the event handler to handle received events thru GCM push
100                KandyCore.getInstance().getRouting().getPushService().setEventHandler(handler);
101                // set the event handler to handle received events thru
102                // IKandyEventsCoreService.getPendingEvents(KandyPendingEventsListner)
103                KandyCore.getInstance().getRouting().getEventsService().setEventHandler(handler);
104                // set the event handler to handle received events from the web socket
105                KandyWebSocketConnectionManager.getInstance().setEventHandler(handler);
106                
107                // generate native device ID and store it locally
108                String nativeDeviceId = KandyDeviceManager.getInstance().getDeviceInfo().getNativeDeviceId();
109                // store it in user session
110                KandyCore.getInstance().getSession().getKandyUser().setNativeDeviceId(nativeDeviceId);
111        }
112
113        public static Context getApplicationContext()
114        {
115                return Kandy.sContext;
116        }
117
118        public static IKandySession getSession()
119        {
120                return KandyCore.getInstance().getSession();
121        }
122
123        public static IKandyProvisioning getProvisioning()
124        {
125                return KandyCore.getInstance().getProvisioning();
126        }
127
128        public static IKandyAccess getAccess()
129        {
130                return KandyCore.getInstance().getAccess();
131        }
132
133        public static IKandyServices getServices()
134        {
135                return KandyCore.getInstance().getServices();
136        }
137        
138        public static IKandyGlobalSettings getGlobalSettings() {
139                return KandyCore.getInstance().getGlobalSettings();
140        }
141        
142        public static String getVersion() {
143                return KandyConfig.KANDY_SDK_VERSOIN_NAME;
144        }
145}
